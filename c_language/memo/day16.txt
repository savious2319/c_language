1. 구조체란?

	여러 타입의 변수를 갖고 있는 사용자 정의 타입

2. 구조체 선언

	#include~
	//구조체 선언!
	키워드  만들고싶은영어단어
	{

	
	}; <--- 여기까지 구조체 선언이야~	
	
	struct MyType{
	           //변수들을 나열한다 : ch, i, d : "멤버"변수
		char ch;
		int i;
		double d;
	};
	struct MyType이라는 자료형을 만들었다!!
	그럼 메인함수에서는 int, char, float, double..... 
	그리고 struct MyType도 자료형으로 쓰일 수 있다! 유효메모리
	함수(){
	}
	
	main(){
	
	}
	
3. 구조체 정의

	main(){
		struct MyType stM;   //stM 이라는 구조체 변수를 만들었다!
	
		//멤버변수를 호출해 보자!
		stM. <--- 멤버변수들이 주~~~욱 나열된다

		stM.ch = 'A'; 멤버변수에 데이터를 넣을 수 있다.
		stM.i = 10;
		stM.d = 1.1;

		.( 점 연산자) : 멤버 직접호출 연산자
			     ~의 00을 호출해라!

		//&stM.i  점이 우선수위. i의 주소값

	}

4. 구조체 초기화

	struct MyType stM = { 0 };
	struct MyType stM = { 'A', 1.1, 10 };  //값손실이 발생되거나 오작동을 일으킬 수 있다.
